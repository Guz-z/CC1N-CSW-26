                    function resolverEquacaoSegundoGrau(a, b, c) {
    if (a === 0) {
        console.log("Não é uma equação de segundo grau, pois a = 0.");
        return;
    }

    // Calcula o discriminante
    const delta = b * b - 4 * a * c;

    if (delta > 0) {
        // Duas raízes reais e diferentes
        const x1 = (-b + Math.sqrt(delta)) / (2 * a);
        const x2 = (-b - Math.sqrt(delta)) / (2 * a);
        console.log(`Duas raízes reais: x1 = ${x1}, x2 = ${x2}`);
    } else if (delta === 0) {
        // Uma raiz real dupla
        const x = -b / (2 * a);
        console.log(`Uma raiz real dupla: x = ${x}`);
    } else {
        // Raízes complexas
        const realPart = -b / (2 * a);
        const imaginaryPart = Math.sqrt(-delta) / (2 * a);
        console.log(`Duas raízes complexas: x1 = ${realPart} + ${imaginaryPart}i, x2 = ${realPart} - ${imaginaryPart}i`);
    }
}

// Exemplo de uso:
resolverEquacaoSegundoGrau(1, -3, 2);  // x^2 - 3x + 2 = 0
resolverEquacaoSegundoGrau(1, 2, 1);   // x^2 + 2x + 1 = 0
resolverEquacaoSegundoGrau(1, 1, 1);   // x^2 + x + 1 = 0
